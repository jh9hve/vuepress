          # HLS MPEGDASH

          https://qiita.com/MMaru76/items/e2495a87b871bc9cbefe

          # nginx.conf

          rtmp {
   server {
      listen 1935;
      chunk_size 4096;

      application live {
         live on;
         record off;

         # HLSの記述欄
         hls on;
         hls_path /var/www/html/hls;
         hls_fragment 10s;

         # MEPG-DASHの記述欄
         dash on;
         dash_path /var/www/html/dash;
         dash_fragment 10s;

      }
   }
}

          html の下に追加

              server {

        }
        location /dash {
            types {
                 application/vnd.apple.mpegurl mpd;
            }
            root /var/www/html/;
        }
    }
